# Create Database
CREATE DATABASE PassagensAereas;
USE PassagensAereas;

# Create tables
CREATE TABLE IF NOT EXISTS Assento
(
    numero INT NOT NULL,
    classe INT NOT NULL,
    n_aeronave INT NOT NULL,
    c_reserva INT NOT NULL,
    PRIMARY KEY(numero)
);

CREATE TABLE IF NOT EXISTS Aeronave
(
    numero_serie INT NOT NULL,
    modelo SMALLINT NOT NULL,
    qtd_assentos INT NOT NULL,
    c_companhia INT NOT NULL,
    PRIMARY KEY(numero_serie)
);

CREATE TABLE IF NOT EXISTS Companhia
(
    codigo INT NOT NULL,
    registro VARCHAR(30) NOT NULL,
    nome VARCHAR(50) NOT NULL,
    PRIMARY KEY(codigo)
);

CREATE TABLE IF NOT EXISTS Voo
(
    codigo INT NOT NULL,
    origem VARCHAR(3) NOT NULL,
    destino VARCHAR(3) NOT NULL,
    data_hora DATETIME NOT NULL,
    n_aeronave INT NOT NULL,
    v_codigo INT NOT NULL,
    PRIMARY KEY(codigo)
);

CREATE TABLE IF NOT EXISTS Aeroporto
(
    sigla VARCHAR(3) NOT NULL,
    local VARCHAR(30) NOT NULL,
    nome VARCHAR(30) NOT NULL,
    PRIMARY KEY(sigla)
);

CREATE TABLE IF NOT EXISTS Reserva
(
    codigo INT NOT NULL,
    status VARCHAR(30) NOT NULL,
    c_passageiro INT NOT NULL,
    v_codigo INT NOT NULL,
    PRIMARY KEY(codigo)
);

CREATE TABLE IF NOT EXISTS Passageiro
(
    codigo INT NOT NULL,
    documento INT NOT NULL,
    nome VARCHAR(50) NOT NULL,
    endereco VARCHAR(50) NOT NULL,
    PRIMARY KEY(codigo)
);

CREATE TABLE IF NOT EXISTS Viagem
(
    codigo INT NOT NULL,
    origem VARCHAR(30),
    destinho VARCHAR(30),
    PRIMARY KEY(codigo)
);

CREATE TABLE IF NOT EXISTS Carrinho
(
    codigo INT NOT NULL,
    p_codigo INT NOT NULL,
    data DATETIME NOT NULL,
    v_codigo INT NOT NULL,
    PRIMARY KEY(codigo, p_codigo)
);


# Create FKs
ALTER TABLE Assento
    ADD    FOREIGN KEY (n_aeronave)
    REFERENCES Aeronave(numero_serie)
;
    
ALTER TABLE Aeronave
    ADD    FOREIGN KEY (c_companhia)
    REFERENCES Companhia(codigo)
;
    
ALTER TABLE Voo
    ADD    FOREIGN KEY (n_aeronave)
    REFERENCES Aeronave(numero_serie)
;
    
ALTER TABLE Assento
    ADD    FOREIGN KEY (c_reserva)
    REFERENCES Reserva(codigo)
;
    
ALTER TABLE Reserva
    ADD    FOREIGN KEY (c_passageiro)
    REFERENCES Passageiro(codigo)
;
    
ALTER TABLE Voo
    ADD    FOREIGN KEY (origem)
    REFERENCES Aeroporto(sigla)
;
    
ALTER TABLE Voo
    ADD    FOREIGN KEY (destino)
    REFERENCES Aeroporto(sigla)
;
    
ALTER TABLE Carrinho
    ADD    FOREIGN KEY (v_codigo)
    REFERENCES Viagem(codigo)
;
    
ALTER TABLE Reserva
    ADD    FOREIGN KEY (v_codigo)
    REFERENCES Viagem(codigo)
;
    
ALTER TABLE Carrinho
    ADD    FOREIGN KEY (p_codigo)
    REFERENCES Passageiro(codigo)
;
    
ALTER TABLE Voo
    ADD    FOREIGN KEY (v_codigo)
    REFERENCES Viagem(codigo)
